-- Creating the Profile table inside the new schema

-- Production Log Table
CREATE TABLE valampure_erp.Manufacturing_Entry (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL REFERENCES valampure_erp.profiles(id),
    production_date DATE NOT NULL DEFAULT CURRENT_DATE,
    
    -- Finished Goods Output
    description TEXT NOT NULL, -- e.g., "White Elastic"
    size_mm TEXT NOT NULL,      -- e.g., "8.5 MM"
    boxes INTEGER NOT NULL,
    total_mts NUMERIC NOT NULL,
    
    -- Raw Material Consumption
    yarn_used_kg NUMERIC NOT NULL DEFAULT 0,
    rubber_used_kg NUMERIC NOT NULL DEFAULT 0,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    CONSTRAINT fk_profile FOREIGN KEY (profile_id) REFERENCES valampure_erp.profiles(id)
);

-- Index for faster searching by date
CREATE INDEX idx_production_date ON valampure_erp.Manufacturing_Entry(production_date);

--==============================================================================================
-- USE: To store the real-time balance of finished elastic rolls/meters.
-- WHEN: Updated automatically whenever a production entry or a sales invoice is saved.
CREATE TABLE valampure_erp.Finished_Goods_Stock (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL REFERENCES valampure_erp.profiles(id),
    description TEXT NOT NULL, -- e.g., "White Elastic"
    size_mm TEXT NOT NULL,      -- e.g., "8.5 MM"
    total_boxes_in_hand INTEGER DEFAULT 0,
    total_mts_in_hand NUMERIC DEFAULT 0,
    last_updated TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- Ensures we don't have duplicate rows for the same size/type
    UNIQUE(profile_id, description, size_mm)
);

-- USE: To track the "In/Out" movement of every specific size of elastic.
-- WHEN: Automatically triggered by Production Logs or Sales Invoices.
CREATE TABLE valampure_erp.Stock_Ledger (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL REFERENCES valampure_erp.profiles(id),
    transaction_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    -- Item Details
    description TEXT NOT NULL, 
    size_mm TEXT NOT NULL,      -- Crucial: "8.5 MM", "20 MM", etc.
    
    -- Movement
    transaction_type TEXT NOT NULL, -- 'PRODUCTION', 'SALE', 'RETURN', 'ADJUSTMENT'
    reference_id UUID,              -- Links back to the specific Production Log or Invoice
    qty_in NUMERIC DEFAULT 0,
    qty_out NUMERIC DEFAULT 0,
    
    -- The balance at the time of this specific transaction
    running_balance NUMERIC NOT NULL, 	
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Index for fast searching by size and date
CREATE INDEX idx_ledger_size ON valampure_erp.Stock_Ledger(size_mm);
CREATE INDEX idx_ledger_date ON valampure_erp.Stock_Ledger(transaction_date);

CREATE TABLE valampure_erp.partners (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL,
    
    -- Identity Details
    name VARCHAR(255) NOT NULL,
    partner_type VARCHAR(20) NOT NULL, -- 'SUPPLIER' or 'BUYER'
    mobile_number VARCHAR(15),
    gstin VARCHAR(15),
    address TEXT,
    state_code VARCHAR(2), -- '33' for Tamil Nadu
    
    -- Financial Tracking
    opening_balance NUMERIC(15, 2) DEFAULT 0.00,
    current_balance NUMERIC(15, 2) DEFAULT 0.00,
    
    -- Metadata
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,

    -- Constraints
    CONSTRAINT fk_profile FOREIGN KEY (profile_id) REFERENCES valampure_erp.profiles(id) ON DELETE CASCADE,
    CONSTRAINT check_partner_type CHECK (partner_type IN ('SUPPLIER', 'BUYER'))
);

-- 2. Add an Index for faster searching by Name and Type
CREATE INDEX idx_partners_name_type ON valampure_erp.partners (name, partner_type);
CREATE INDEX idx_partners_profile ON valampure_erp.partners (profile_id);

-- 3. Create a Trigger to update 'updated_at' automatically
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ language 'plpgsql';

CREATE TRIGGER update_partners_modtime
    BEFORE UPDATE ON valampure_erp.partners
    FOR EACH ROW
    EXECUTE PROCEDURE update_updated_at_column();

-- 1. THE MASTER TABLE (The Bill Header)
CREATE TABLE valampure_erp.purchase_master (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL,
    partner_id UUID NOT NULL, -- Links to Jai Traders
    
    bill_number VARCHAR(50),
    bill_date DATE DEFAULT CURRENT_DATE,
    
    -- Storage & Tax Info
    bill_url TEXT,            -- MinIO Link to the scanned bill
    is_gst BOOLEAN DEFAULT FALSE,
    
    -- Financials (The Final Truth)
    total_sub_total NUMERIC(15, 2) DEFAULT 0.00,
    total_tax_amount NUMERIC(15, 2) DEFAULT 0.00,
    final_amount NUMERIC(15, 2) NOT NULL, -- The EDITABLE field from your UI
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_profile FOREIGN KEY (profile_id) REFERENCES valampure_erp.profiles(id),
    CONSTRAINT fk_partner FOREIGN KEY (partner_id) REFERENCES valampure_erp.partners(id)
);

-- 2. THE DETAIL TABLE (The Item Categories)
CREATE TABLE valampure_erp.purchase_details (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    purchase_id UUID NOT NULL REFERENCES valampure_erp.purchase_master(id) ON DELETE CASCADE,
    
    category VARCHAR(50) NOT NULL, -- Yarn, Rubber, Boxes, etc.
    quantity NUMERIC(15, 2) NOT NULL,
    uom VARCHAR(10) NOT NULL,      -- Kgs, Pcs, Rolls
    rate NUMERIC(15, 2) NOT NULL,
    line_total NUMERIC(15, 2) NOT NULL, -- Qty * Rate
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 1. Create the Function
CREATE OR REPLACE FUNCTION valampure_erp.fn_update_partner_balance_on_purchase()
RETURNS TRIGGER AS $$
BEGIN
    -- Increase the partner's balance by the final_amount of the purchase
    UPDATE valampure_erp.partners
    SET current_balance = current_balance + NEW.final_amount
    WHERE id = NEW.partner_id;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Create the Trigger
CREATE TRIGGER trg_after_purchase_insert
AFTER INSERT ON valampure_erp.purchase_master
FOR EACH ROW
EXECUTE FUNCTION valampure_erp.fn_update_partner_balance_on_purchase();

CREATE OR REPLACE FUNCTION valampure_erp.fn_update_stock_on_sale()
RETURNS TRIGGER AS $$
DECLARE
    v_stock_id UUID;
BEGIN
    -- Match by BOTH size and description
    SELECT id INTO v_stock_id
    FROM valampure_erp.finished_goods_stock
    WHERE description = NEW.description 
      AND size_mm = NEW.size_mm  -- This is the missing link!
      AND profile_id = (SELECT profile_id FROM valampure_erp.sales_master WHERE id = NEW.sales_id);

    IF v_stock_id IS NULL THEN
        RAISE EXCEPTION 'No matching stock found for: % (Size: %)', NEW.description, NEW.size_mm;
    END IF;

    -- Update the stock
    UPDATE valampure_erp.finished_goods_stock
    SET total_boxes_in_hand = total_boxes_in_hand - NEW.box_count,
        total_mts_in_hand = total_mts_in_hand - NEW.total_qty,
        last_updated = CURRENT_TIMESTAMP
    WHERE id = v_stock_id;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_after_sales_item_insert
AFTER INSERT ON valampure_erp.sales_detail
FOR EACH ROW
EXECUTE FUNCTION valampure_erp.fn_update_stock_on_sale();

CREATE TABLE valampure_erp.payments (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    profile_id UUID NOT NULL, -- Links to your Valampure Profile
    partner_id UUID NOT NULL REFERENCES valampure_erp.partners(id) ON DELETE CASCADE,
    
    payment_date DATE NOT NULL DEFAULT CURRENT_DATE,
    -- 'RECEIPT' for money from Buyers, 'PAYMENT' for money to Suppliers
    payment_type TEXT NOT NULL CHECK (payment_type IN ('RECEIPT', 'PAYMENT')), 
    
    amount NUMERIC(15, 2) NOT NULL CHECK (amount > 0),
    payment_mode TEXT, -- e.g., 'Cash', 'UPI', 'Bank Transfer'
    reference_no TEXT, -- e.g., Transaction ID or Cheque Number
    remarks TEXT,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Index for faster ledger lookups
CREATE INDEX idx_payments_partner_id ON valampure_erp.payments(partner_id);

-- 1. Create the Function
CREATE OR REPLACE FUNCTION valampure_erp.fn_update_partner_balance_on_payment()
RETURNS TRIGGER AS $$
BEGIN
    -- Subtract the payment amount from the partner's balance
    -- (Reduces Payable for Suppliers and Receivable for Buyers)
    UPDATE valampure_erp.partners
    SET current_balance = current_balance - NEW.amount
    WHERE id = NEW.partner_id;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Create the Trigger
CREATE TRIGGER trg_after_payment_insert
AFTER INSERT ON valampure_erp.payments
FOR EACH ROW
EXECUTE FUNCTION valampure_erp.fn_update_partner_balance_on_payment();
